Configuration 1/2


# widget nginx + memcached
location /widget/ {
    set $memcached_key $geoip_city$uri;

    if ($arg_company != '') {
        set $memcached_key $memcached_key?company=$arg_company
    }

    default_type text/html;
    memcached_pass 127.0.0.1:11211;
    error_page 404 = @fallback;
    expires +7200;
}
